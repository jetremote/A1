014package com.jetremote.xml;
015
016/**
017 * Model class. Its instances will be populated using SAX parser. 
018 * */
019public class ModuleRF 
020{
021        //XML attribute id
022        private String id;
023        //XML element serialNumberHigh
024        private String serialHigh;
025        //XML element serialNumberLow
026        private String serialLow;
027        //XML element model
028        private String model;
029        
030        public String getId() {
031                return id;
032        }
033        public void setId(String id) {
034                this.id = id;
035        }
036        
037        public String getSerialHigh() {
038                return serialHigh;
039        }
040        public void setSerialHigh(String serialHigh) {
041                this.serialHigh = serialHigh;
042        }
043        public String getSerialLow() {
044                return serialLow;
045        }
046        public void setSerialLow(String serialLow) {
047                this.serialLow = serialLow;
048        }
049        public String getFullAddress(){
050                return serialHigh+serialLow;
051        }
052        public String getModel() {
053                return model;
054        }
055        public void setModel(String model) {
056                this.model = model;
057        }
058        @Override
059        public String toString() {
060                return this.id + ":" + this.serialHigh + this.serialLow ;
061        }
062}
