014package com.jetremote.xml;
015
016
017import java.io.IOException;
018import java.io.InputStream;
019import java.util.ArrayList;
020
021import org.xml.sax.InputSource;
022import org.xml.sax.SAXException;
023import org.xml.sax.XMLReader;
024import org.xml.sax.helpers.XMLReaderFactory;
025
026public class ModuleRFXmlParser 
027{
028        public ArrayList<ModuleRF> parseXml(InputStream in)
029        {
030                //Create a empty link of modules initially
031                ArrayList<ModuleRF> modules = new ArrayList<ModuleRF>();
032                try 
033                {
034                        //Create default handler instance
035                        ModuleRFParserHandler handler = new ModuleRFParserHandler();
036                        
037                        //Create parser from factory
038                        XMLReader parser = XMLReaderFactory.createXMLReader();
039                        
040                        //Register handler with parser
041                        parser.setContentHandler(handler);
042
043                        //Create an input source from the XML input stream
044                        InputSource source = new InputSource(in);
045                        
046                        //parse the document
047                        parser.parse(source);
048                        
049                        //populate the parsed modules list in above created empty list
050                        modules = handler.getModules();
051
052                } catch (SAXException e) {
053                        e.printStackTrace();
054                } catch (IOException e) {
055                        e.printStackTrace();
056                } finally {
057
058                }
059                return modules;
060        }
061}
